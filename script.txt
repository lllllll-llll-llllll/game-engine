echo on
do routine1
do routine2
ran random 1 100
say name [random]
set gold 24
set hp 14

if [gold] = ([hp] + 10)
	y say name yes.
	y say name gold is 34.
	if [hp] = 14
		y say name hp is 14.
		n say name hp isn't 14.
	end
	n say name no.
	n say name gold isn't 34.
end

case [gold]
	24 say name gold is 24.
	10 say name gold is 10.
end

to routine1
	do routine2
	say name this is routine1.
return

to routine2
	say name this is routine2 called from within routine1.
return

to routine3
	say name this is routine3.
return

wait 100
exit
